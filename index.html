1.单元素修改的线段树

  • 我们这里考虑两种操作

      1. 修改数组中的一个元素
      2. 求某个区间的元素总和

    • 这里我们先假设数组的元素个数为2的幂，这里以16为例，最下面一行（叶子节点）是我们的数组

      • 修改操作的话就直接修改然后向上维护就行
        求和操作：
        举个栗子，我们需要求[3,11)的和，只需要对19、5、12、26节点求和即可

      • #include<bits/stdc++.h>

using namespace std;

const int N = 1e6 + 5;

int n, t[N << 1];

//build the tree
void build() {
   for(int i = n - 1;i > 0; --i) t[i] = t[i<<1] + t[i<<1|1];
}

//set val at position p
void modify(int p,int val) {
   for(t[p += n] = val; p > 1; p >>= 1) t[p>>1] = t[p] + t[p^1];
}

//sum on interval [l, r)
int query(int l,int r) {
   int res = 0;
   for(l += n, r += n; l < r;l >>= 1, r >>= 1) {
       if(l&1) res += t[l++];
       if(r&1) res += t[--r];
   }
   return res;
}

int main(){
   scanf("%d",&n);
   for(int i = 0; i < n; ++i) scanf("%d", t + n + i);
   build();modify(0,1);
   printf("%d\n",query(3,11));
   return 0;
}

void modify(int l, int r, int value) {
 for (l += n, r += n; l < r; l >>= 1, r >>= 1) {
   if (l&1) t[l++] += value;
   if (r&1) t[--r] += value;
 }
}

int query(int p) {
 int res = 0;
 for (p += n; p > 0; p >>= 1) res += t[p];
 return res;
}

    • 但是请注意，以上代码仅在单个元素的修改顺序不影响结果的情况下才有效。例如，分配不满足此条件。有关更多信息，请参阅关于延迟传播的部分。 （就是lazy标记啦）
